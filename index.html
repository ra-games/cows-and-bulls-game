    <div class="game-container">

        <h1>Cows and Bulls Game</h1>

        <p>Guess the 4-letter word!</p>

        

        <input type="text" id="userGuess" maxlength="4" placeholder="Enter a word">

        <button onclick="checkGuess()">Submit</button>

        

        <div id="result"></div>

    </div>


    <script>

        let validWords = []; 

        let secretWord = "";


        async function loadWords() {

            try {

                const response = await fetch("words.txt");

                const text = await response.text();

                validWords = text.split("\n").map(word => word.trim().toLowerCase()).filter(word => word.length === 4);

                secretWord = validWords[Math.floor(Math.random() * validWords.length)];

            } catch (error) {

                console.error("Error loading words:", error);

            }

        }


        function checkGuess() {

            let userGuess = document.getElementById("userGuess").value.toLowerCase();

            if (!validWords.includes(userGuess)) {

                document.getElementById("result").innerText = "Invalid word! Choose from the list.";

                return;

            }


            let cows = 0, bulls = 0;

            for (let i = 0; i < 4; i++) {

                if (userGuess[i] === secretWord[i]) {

                    bulls++;

                } else if (secretWord.includes(userGuess[i])) {

                    cows++;

                }

            }


            if (bulls === 4) {

                document.getElementById("result").innerText = `🎉 Correct! The word was "${secretWord}"`;

            } else {

                document.getElementById("result").innerText = `Cows: ${cows}, Bulls: ${bulls}`;

            }

        }


        loadWords();

    </script>
